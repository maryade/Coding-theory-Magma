/*
Code for generating the first 20 lines of Pascal's Triangle */


S:=[];

S[1]:=[1];

S[2]:=[1,1];

for i in [3..20] do

S[i]:=[];

S[i][1]:=1;

for j in [2..i-1] do

S[i][j] := S[i-1][j-1]+S[i-1][j];

end for;

S[i][i]:=1;

end for;

S;
